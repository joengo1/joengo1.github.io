#!/bin/java

public class quicksort
{
  public static void main(String[] args)
  {
    int[] arr = {4, 5, 23, 42, 1,  14, 6, 33, 24, 25, 53, 10, 9};

    quicksort(arr, 0, arr.length - 1);

    for(int element: arr){
      System.out.print(element + " ");
    }
    System.out.println();
  }

  public static void quicksort(int[] a, int p, int r)
  {
    if(p < r){
      int q = partition(a, p, r);
      quicksort(a, p, q - 1);
      quicksort(a, q + 1, r);
    }
  }

  public static int partition(int[]a, int p, int r){
    int x, i, temp;
    x = a[r];
    i = p - 1;

    for(int j = p; j <= r - 1; j++){
      if(a[j] <= x){
        i++;
        temp = a[i];
        a[i] = a[j];
        a[j] = temp;
      }
    }

    temp = a[i + 1];
    a[i + 1] = a[r];
    a[r] = temp;

    return i + 1;
  }
}
