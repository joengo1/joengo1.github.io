#!/bin/java

public class binary_search
{
  public static void main(String[] args)
  {
    int[] arr = {1, 23, 45, 65, 113, 150, 200, 250, 264, 304, 345};

    int value = 345;
    int value_idx = binary_search(arr, value);

    if(value_idx == -1){
      System.out.println("Value being searched does not exist");
    }
    else{
      System.out.println("Value " + value + " is located at index: " + value_idx);
    }
  }

  // Searches sorted array O(nlgn)
  public static int binary_search(int[] a, int value){
    int first = 0;
    int last = a.length;

    while(first < last){
      int middle = (int)Math.floor((first + last) / 2);
      if(value <= a[middle]){
        last = middle;
      }
      else{
        first = middle + 1;
      }
    }

    if(a[first] == value){
      return first;
    }
    else{
      return -1;
    }
  }
}
