#!/bin/java

import java.util.*;
import java.lang.*;
import java.io.*;

public class fib
{
  public static void main(String[] args)
  {
    List fib_list = new ArrayList();

    fib_list.add(0);
    fib_list.add(1);

    fib_list = fib_compute(fib_list, 2, 100);

    for(int element: fib_list){
      System.out.println(element);
    }
  }

  public static List fib_compute(List list, int idx, int end){
    if(idx == end){
      return list;
    }
    else{
      list.add(list.get(idx - 2) + list.get(idx - 1));
      fib_compute(list, ++idx, end);
    }

    return list;
  }
}
