#!/bin/java

public class factorial
{
  public static void main(String[] args)
  {
      int num = 5;
      int result = fact(num);

      System.out.println("Factorial(" + num + ") = " + result);
  }


  public static int fact(int n){
    int f = n;

    if(n < 0){
      System.out.println("Error: number is less than 0");
    }
    else if(n == 0){
      f = 1;
    }
    else{
      f = n * fact(n - 1);
    }

    return f;
  }
}
