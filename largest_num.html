#!/bin/java

import java.util.*;

public class largest_num
{
  public static void main(String[] args)
  {
    int[] num_arr = {50, 2, 1, 9, 800, 980, 9323};
    int[] sorted;
    String largest = "";

    sorted = counting_sort(num_arr);

    for(int element: sorted){
      System.out.println(element);
    }
    System.out.println();

    for(int i = sorted.length - 1; i >= 0; i--){
      largest += String.valueOf(sorted[i]);
    }

    System.out.println(largest);
  }

  public static int[] counting_sort(int[] arr){
    int[] c = new int[10];
    int[] b = new int[arr.length];

    // Count occurences of 1 ... k
    for(int element: arr){
      int digit = Character.getNumericValue(String.valueOf(element).charAt(0));
      c[digit]++;
    }

    for(int element: c){
        System.out.println(element);
    }
    System.out.println();

    for(int i = 1; i < c.length; i++){
      c[i] = c[i-1] + c[i];
    }

    for(int element: c){
        System.out.println(element);
    }
    System.out.println();

    for(int i = arr.length - 1; i >= 0; i--){
      int digit = Character.getNumericValue(String.valueOf(arr[i]).charAt(0));
      b[c[digit] - 1] = arr[i];
      c[digit]--;
    }

    return b;
  }
}
