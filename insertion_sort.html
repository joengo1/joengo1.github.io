#!/bin/java

public class insertion_sort
{
  public static void main(String[] args)
  {
    int[] arr = {45, 2, 5, 12, 10, -1, -50, 60, 23};

    insertion_sort(arr);

    for(int element: arr){
      System.out.print(element + " ");
    }
    System.out.println();
  }

  // Insert Sort O(n^2)
  public static void insertion_sort(int[] a){
    for(int j = 1; j < a.length; j++){
      int key = a[j];
      int i = j - 1;

      while(i >= 0 && a[i] > key){
        a[i + 1] = a[i];
        i--;
      }

      a[i + 1] = key;
    }
  }
}
