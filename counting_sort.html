#!/bin/java

public class counting_sort
{
  public static void main(String[] args)
  {
    int[] arr = {56, 4, 2, 34, 1, 4, 5, 33, 5, 2, 4, 6, 2, 3, 45, 34, 1, 56};

    int[] sorted_arr = counting_sort(arr);

    for(int element: sorted_arr){
      System.out.print(element + " ");
    }
    System.out.println();
  }

  public static int[] counting_sort(int[] a){

		int[] c = new int[max(a) + 1];
		int[] b = new int[a.length];

		// Count occurences of 1 ... k in arr
		for(int element: a){
		  c[element]++;
		}

		for(int i = 1; i < c.length; i++){
		  c[i] = c[i-1] + c[i];
		}

		for(int i = a.length - 1; i >= 0; i--){
		  int idx = a[i];
		  b[c[idx] - 1] = a[i];
		  c[idx]--;
		}

    return b;
  }

  public static int max(int[] a){
  	int max = a[0];

    for(int i = 1; i < a.length; i++)
    {
        if(a[i] > max){
        	max = a[i];
        }
    }

    return max;
  }
}
